# Конфигурация Nginx для CF Platform
# Сохраните как /etc/nginx/sites-available/cf-platform

server {
    listen 80;
    server_name YOUR_DOMAIN_OR_IP;

    # Логи
    access_log /var/log/nginx/cf-platform.access.log;
    error_log /var/log/nginx/cf-platform.error.log;

    # Проксирование к Node.js приложению
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }

    # Статические файлы
    location /static/ {
        alias /home/CC_Platform/public/;
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
}

# Команды для активации:
# sudo ln -s /etc/nginx/sites-available/cf-platform /etc/nginx/sites-enabled/
# sudo nginx -t
# sudo systemctl reload nginx
